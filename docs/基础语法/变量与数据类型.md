# 变量与数据类型

变量是程序中用来存储数据的基本单位，而数据类型决定了变量可以存储什么类型的数据以及如何存储。

## 基本数据类型

### 整型
```c
#include <stdio.h>

int main() {
    // 不同大小的整型
    short int a = 10;        // 短整型 (通常2字节)
    int b = 100;             // 整型 (通常4字节)
    long int c = 1000;       // 长整型 (通常4或8字节)
    long long d = 10000;     // 长长整型 (通常8字节)

    // 无符号整型
    unsigned int e = 200;    // 无符号整型
    unsigned long f = 3000;  // 无符号长整型

    printf("short int: %d\n", a);
    printf("int: %d\n", b);
    printf("long int: %ld\n", c);
    printf("long long: %lld\n", d);
    printf("unsigned int: %u\n", e);
    printf("unsigned long: %lu\n", f);

    return 0;
}
```

### 字符型
```c
#include <stdio.h>

int main() {
    char ch1 = 'A';          // 字符
    char ch2 = 65;           // 字符的ASCII码
    signed char ch3 = -100;  // 有符号字符
    unsigned char ch4 = 200; // 无符号字符

    printf("字符: %c\n", ch1);
    printf("ASCII码对应的字符: %c\n", ch2);
    printf("字符的ASCII码: %d\n", ch1);
    printf("有符号字符: %d\n", ch3);
    printf("无符号字符: %d\n", ch4);

    return 0;
}
```

### 浮点型
```c
#include <stdio.h>

int main() {
    float f1 = 3.14f;        // 单精度浮点数
    double d1 = 3.141592653; // 双精度浮点数

    printf("float: %f\n", f1);
    printf("double: %lf\n", d1);
    printf("科学计数法(float): %e\n", f1);
    printf("科学计数法(double): %e\n", d1);

    return 0;
}
```

```c

int main() {
```

## 变量的声明和初始化

### 声明方式
```c
#include <stdio.h>

int main() {
    // 先声明，后赋值
    int age;
    age = 20;

    // 声明时初始化
    int height = 175;
    float weight = 65.5f;
    char grade = 'A';

    // 同时声明多个变量
    int x, y, z;
    int a = 1, b = 2, c = 3;

    printf("年龄: %d\n", age);
    printf("身高: %d cm\n", height);
    printf("体重: %.1f kg\n", weight);
    printf("成绩等级: %c\n", grade);

    return 0;
}
```

### 常量定义
```c
#include <stdio.h>

#define PI 3.14159f          // 宏定义常量
const int MAX_SIZE = 100;   // const关键字定义常量

int main() {
    int radius = 5;
    float area = PI * radius * radius;

    printf("圆的面积: %.2f\n", area);
    printf("最大尺寸: %d\n", MAX_SIZE);

    // MAX_SIZE = 200;  // 错误！常量不能修改

    return 0;
}
```

## 数据类型转换

### 自动类型转换
```c
#include <stdio.h>

int main() {
    int i = 5;
    float f = 2.5f;
    double d = 3.14;

    // 自动类型转换（小类型转换为大类型）
    double result1 = i + f;      // int转为float，然后转为double
    double result2 = f + d;      // float转为double

    printf("i + f = %f\n", result1);
    printf("f + d = %f\n", result2);

    return 0;
}
```

### 强制类型转换
```c
#include <stdio.h>

int main() {
    int a = 5, b = 2;
    float result1, result2;

    // 整数除法（结果为整数）
    result1 = a / b;
    printf("5 / 2 = %f\n", result1);  // 输出 2.000000

    // 强制类型转换
    result2 = (float)a / b;
    printf("(float)5 / 2 = %f\n", result2);  // 输出 2.500000

    // 其他类型转换示例
    int i = 3.7;              // double转为int，小数部分被截断
    printf("int i = %d\n", i);  // 输出 3

    return 0;
}
```

## 变量命名规则

### 命名规范
```c
#include <stdio.h>

int main() {
    // 合法的变量名
    int age;                  // 小写字母
    int student_age;          // 下划线分隔
    int studentAge;           // 驼峰命名法
    int age1;                 // 可以包含数字

    // 常量命名（通常大写）
    const int MAX_AGE = 150;

    // 初始化变量
    age = 20;
    student_age = 18;
    studentAge = 19;
    age1 = 25;

    printf("年龄: %d\n", age);
    printf("学生年龄: %d\n", student_age);
    printf("学生年龄(驼峰): %d\n", studentAge);
    printf("年龄1: %d\n", age1);
    printf("最大年龄: %d\n", MAX_AGE);

    return 0;
}
```

### 命名规则总结
1. 只能使用字母、数字和下划线
2. 必须以字母或下划线开头
3. 不能使用关键字作为变量名
4. 区分大小写
5. 建议使用有意义的名称

## 实际应用示例

### 计算圆的周长和面积
```c
#include <stdio.h>

#define PI 3.14159f

int main() {
    float radius, circumference, area;

    printf("请输入圆的半径: ");
    scanf("%f", &radius);

    circumference = 2 * PI * radius;
    area = PI * radius * radius;

    printf("周长: %.2f\n", circumference);
    printf("面积: %.2f\n", area);

    return 0;
}
```

### 温度转换
```c
#include <stdio.h>

int main() {
    float celsius, fahrenheit;

    printf("请输入摄氏温度: ");
    scanf("%f", &celsius);

    fahrenheit = celsius * 9.0f / 5.0f + 32.0f;

    printf("华氏温度: %.1f°F\n", fahrenheit);

    return 0;
}
```
